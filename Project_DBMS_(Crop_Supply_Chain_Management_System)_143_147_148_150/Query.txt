CREATE TABLE Admin (
    ID_No INT PRIMARY KEY,
    Employee_Name VARCHAR(255),
    Contact_No VARCHAR(15),
    Email_Address VARCHAR(255)
);

CREATE TABLE CropDemandInMarket (
    Crop_ID INT PRIMARY KEY,
    ID_No INT,
    Market_Demand INT,
    Price_Per_Unit DECIMAL(10, 2),
    FOREIGN KEY (ID_No) REFERENCES Admin(ID_No)
);

CREATE TABLE Farmer (
    Farmer_ID INT PRIMARY KEY,
    ID_No INT,
    Crop_ID INT,
    Farmer_Name VARCHAR(255),
    Age INT,
    Gender VARCHAR(10),
    Farmer_Contact_Number VARCHAR(15),
    FOREIGN KEY (ID_No) REFERENCES Admin(ID_No),
    FOREIGN KEY (Crop_ID) REFERENCES CropDemandInMarket(Crop_ID)
);
CREATE TABLE Property (
    Property_ID INT PRIMARY KEY,
    Farmer_ID INT,
    Property_Address VARCHAR(255),
    Property_Size DECIMAL(10, 2),
    Property_Value DECIMAL(15, 2),
    Crop_Type VARCHAR(50),
    FOREIGN KEY (Farmer_ID) REFERENCES Farmer(Farmer_ID)
);

CREATE TABLE FarmersCost (
    Cost_ID INT PRIMARY KEY,
    ID_No INT,
    Farmer_ID INT,
    Seed_Cost DECIMAL(10, 2),
    Fertilizer_Cost DECIMAL(10, 2),
    Labor_Cost DECIMAL(10, 2),
    Other_Expenses DECIMAL(10, 2),
    FOREIGN KEY (ID_No) REFERENCES Admin(ID_No),
    FOREIGN KEY (Farmer_ID) REFERENCES Farmer(Farmer_ID)
);

CREATE TABLE HarvestingEquipmentSystem (
    Equipment_ID INT PRIMARY KEY,
    ID_No INT,
    Equipment_Name VARCHAR(255),
    Equipment_Type VARCHAR(50),
    Equipment_Cost DECIMAL(10, 2),
    FOREIGN KEY (ID_No) REFERENCES Admin(ID_No)
);

CREATE TABLE CropBatch (
    Batch_ID INT PRIMARY KEY,
    Crop_ID INT,
    Farmer_ID INT,
    ID_No INT,
    Batch_Size DECIMAL(10, 2),
    Batch_Rate DECIMAL(10, 2),
    Batch_Date DATE,
    FOREIGN KEY (Crop_ID) REFERENCES CropDemandInMarket(Crop_ID),
    FOREIGN KEY (Farmer_ID) REFERENCES Farmer(Farmer_ID),
    FOREIGN KEY (ID_No) REFERENCES Admin(ID_No)
);

CREATE TABLE FarmersProfit (
    Profit_ID INT PRIMARY KEY,
    Farmer_ID INT,
    Total_Revenue DECIMAL(15, 2),
    Total_Expenses DECIMAL(15, 2),
    Net_Profit DECIMAL(15, 2),
    FOREIGN KEY (Farmer_ID) REFERENCES Farmer(Farmer_ID)
);

CREATE TABLE FarmersLoan (
    Loan_ID INT PRIMARY KEY,
    Farmer_ID INT,
    Cost_ID INT,
    ID_No INT,
    Loan_Amount DECIMAL(15, 2),
    Interest_Rate DECIMAL(5, 2),
    Loan_Duration INT,
    FOREIGN KEY (Farmer_ID) REFERENCES Farmer(Farmer_ID),
    FOREIGN KEY (Cost_ID) REFERENCES FarmersCost(Cost_ID),
    FOREIGN KEY (ID_No) REFERENCES Admin(ID_No)
);

CREATE TABLE ProductSupply (
    Supply_ID INT PRIMARY KEY,
    Crop_ID INT,
    ID_No INT,
    Supply_Quantity DECIMAL(10, 2),
    Supply_Date DATE,
    FOREIGN KEY (Crop_ID) REFERENCES CropDemandInMarket(Crop_ID),
    FOREIGN KEY (ID_No) REFERENCES Admin(ID_No)
);

CREATE TABLE SupplyManagement (
    Supplier_ID INT PRIMARY KEY,
    Supply_ID INT,
    Batch_ID INT,
    ID_No INT,
    Supplier_Name VARCHAR(255),
    Management_Type VARCHAR(50),
    FOREIGN KEY (Supply_ID) REFERENCES ProductSupply(Supply_ID),
    FOREIGN KEY (Batch_ID) REFERENCES CropBatch(Batch_ID),
    FOREIGN KEY (ID_No) REFERENCES Admin(ID_No)
);

CREATE TABLE Manufacturer (
    Product_ID INT PRIMARY KEY,
    Supply_ID INT,
    ID_No INT,
    Product_Name VARCHAR(255),
    Product_Price DECIMAL(10, 2),
    No_of_Packages INT,
    FOREIGN KEY (Supply_ID) REFERENCES ProductSupply(Supply_ID),
    FOREIGN KEY (ID_No) REFERENCES Admin(ID_No)
);

CREATE TABLE DistributionCenter (
    DCenter_ID INT PRIMARY KEY,
    Product_ID INT,
    ID_No INT,
    Center_Name VARCHAR(255),
    Center_Location VARCHAR(255),
    FOREIGN KEY (Product_ID) REFERENCES Manufacturer(Product_ID),
    FOREIGN KEY (ID_No) REFERENCES Admin(ID_No)
);

CREATE TABLE ShopsSellerMarket (
    Shop_ID INT PRIMARY KEY,
    Product_ID INT,
    Tax_ID INT,
    DCenter_ID INT,
    Shop_Name VARCHAR(255),
    Shop_Location VARCHAR(255),
    FOREIGN KEY (Product_ID) REFERENCES Manufacturer(Product_ID),
    FOREIGN KEY (Tax_ID) REFERENCES Tax(Tax_ID),
    FOREIGN KEY (DCenter_ID) REFERENCES DistributionCenter(DCenter_ID)
);

CREATE TABLE Exportation (
    Shipment_ID INT PRIMARY KEY,
    Crop_ID INT,
    Tax_ID INT,
    Supplier_ID INT,
    Shipment_Cost DECIMAL(10, 2),
    Export_Quantity DECIMAL(10, 2),
    Export_Date DATE,
    FOREIGN KEY (Crop_ID) REFERENCES CropDemandInMarket(Crop_ID),
    FOREIGN KEY (Tax_ID) REFERENCES Tax(Tax_ID),
    FOREIGN KEY (Supplier_ID) REFERENCES SupplyManagement(Supplier_ID)
);

CREATE TABLE Tax (
    Tax_ID INT PRIMARY KEY,
    ID_No INT,
    Tax_Type VARCHAR(50),
    Tax_Rate DECIMAL(5, 2),
    FOREIGN KEY (ID_No) REFERENCES Admin(ID_No)
);

CREATE TABLE BuyersConsumer (
    Buyers_ID INT PRIMARY KEY,
    Product_ID INT,
    Tax_ID INT,
    Shop_ID INT,
    Buyers_Name VARCHAR(255),
    Buyers_Contact VARCHAR(15),
    FOREIGN KEY (Product_ID) REFERENCES Manufacturer(Product_ID),
    FOREIGN KEY (Tax_ID) REFERENCES Tax(Tax_ID),
    FOREIGN KEY (Shop_ID) REFERENCES ShopsSellerMarket(Shop_ID)
);

CREATE TABLE TotalTransactionInEconomy (
    Transaction_ID INT PRIMARY KEY,
    Tax_ID INT,
    ID_No INT,
    Total_Transaction_Value DECIMAL(15, 2),
    Transaction_Date DATE,
    FOREIGN KEY (Tax_ID) REFERENCES Tax(Tax_ID),
    FOREIGN KEY (ID_No) REFERENCES Admin(ID_No)
);

query:
Retrieve Farmers' Names and Contact Numbers:

SELECT Farmer_Name, Farmer_Contact_Number FROM Farmer;



.Concat Operation:
SELECT concat('ID_No', ' ', 'Email_Address')As new_information;

Count the Number of Shops in Each Location:

SELECT Shop_Location, COUNT(Shop_ID) AS Shop_Count
FROM Shops_Seller_Market
GROUP BY Shop_Location;


Find the Total Transaction Value for Each Tax Type:

SELECT Tax_Type, SUM(Total_Transaction_Value) AS Total_Value
FROM Tax
JOIN Total_Transaction_Economy ON Tax.Tax_ID = Total_Transaction_Economy.Tax_ID
GROUP BY Tax_Type;

List the Manufacturers with their Product Names and Prices:
SELECT Manufacturer.Product_ID, Manufacturer.Product_Name, Manufacturer.Product_Price
FROM Manufacturer;


Retrieve Crop Batches with a Batch Rate Over $10.00:

SELECT Batch_ID, Crop_Name, Batch_Rate
FROM Crop_Batch
WHERE Batch_Rate > 10.00;


.Identify Farmers Who Took Loans with a Loan Amount Over $20,000:

SELECT Farmer_ID, Loan_Amount
FROM Farmers_Loan
WHERE Loan_Amount > 20000;


Find the Average Price Per Unit of Each Crop:
SELECT Crop_ID, AVG(Price_Per_Unit) AS Avg_Price_Per_Unit
FROM Crop_Demand_Market
GROUP BY Crop_ID;



List the Top 5 Farmers by Net Profit:
SELECT Farmer_ID, Net_Profit
FROM Farmers_Profit
ORDER BY Net_Profit DESC
LIMIT 5;



Calculate Total Expenses for Each Farmer:

SELECT Farmer_ID, SUM(Seed_Cost + Fertilizer_Cost + Labor_Cost + Other_Expenses) AS Total_Expenses
FROM Farmers_Cost GROUP BY Farmer_ID;



Find the Total Supply Quantity of Each Product:

SELECT Supply_ID, SUM(Supply_Quantity) AS Total_Supply_Quantity
FROM Product_Supply
